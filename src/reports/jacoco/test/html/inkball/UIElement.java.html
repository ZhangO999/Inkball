package inkball;

import processing.core.PApplet;

public abstract class UIElement {
    protected int x, y;
    protected int width, height; // Size of the element

    public UIElement(int x, int y, int width, int height) {
        this.x = x;
        this.y = y;
        this.width = width;
        this.height = height;
    }

    public abstract void draw(PApplet app);
}
